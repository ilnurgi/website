    .. code-block:: py

        from Tkinter import Tk, Text, Scrollbar

        root = Tk()
        text = Text(root, height=3, width=60)
        text.pack(side='left')

        scrollbar = Scrollbar(root)
        scrollbar.pack()

        # первая привязка
        scrollbar['command'] = text.yview

        # вторая привязка
        text['yscrollcommand'] = scrollbar.set

        root.mainloop()
